# Use official Node LTS
FROM node:18-alpine

# Create app directory
WORKDIR /usr/src/app

# Install dependencies
COPY package*.json ./
RUN npm ci --only=production

# Copy source
COPY . .

# Create a writable directory for SQLite DB
RUN mkdir -p /data
VOLUME ["/data"]

# Environment defaults
ENV PORT=8765
ENV DB_FILE=/data/cards.db

# Expose port
EXPOSE 8765

# Start the server
CMD ["npm", "start"]
